def hash_password(self, password: str, salt: bytes) -> str:
    key = hashlib.pbkdf2_hmac(
        'sha256',
        password.encode('utf-8'),
        salt,
        100000
    )
    return key.hex()

def register_user(self, email: str, password: str):
    salt = os.urandom(32)
    pwd_hash = self.hash_password(password, salt)
    self.db[email] = {
        "salt": salt.hex(),
